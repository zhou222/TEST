猜数字游戏的程序：

#include<stdio.h>
int main(void)
{
    int guess=1;
    int ch;
    printf("is it %d?\n",guess);
    while((ch=getchar())!='y')
    {
        if(ch=='n')
          printf("is it %d?\n",++guess);
        else
          printf("sorry!i can`t understand!i only can understand 'n'or'y'.\n");
        while(getchar()!='\n')
           continue;
    }
    printf("i have got it");
}
